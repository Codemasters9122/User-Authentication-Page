<script>
    // Prevent copy paste in password fields
    var passwordFields = document.querySelectorAll('input[type="password"]');
    passwordFields.forEach(function (passwordField) {
        passwordField.addEventListener('copy', function (event) {
            event.preventDefault();
        });
        passwordField.addEventListener('paste', function (event) {
            event.preventDefault();
        });
    });

    // Password strength meter
    document.getElementById('password').addEventListener('input', function () {
        var password = document.getElementById('password').value;
        var strengthBadge = document.getElementById('password-strength');
        var strengthMeter = document.getElementById('password-strength').querySelector('::before');
        var passwordField = document.getElementById('password');

        var strength = 0;
        if (password.match(/[a-z]+/)) {
            strength += 1;
        }
        if (password.match(/[A-Z]+/)) {
            strength += 1;
        }
        if (password.match(/[0-9]+/)) {
            strength += 1;
        }
        if (password.length >= 8) {
            strength += 1;
        }

        switch (strength) {
            case 0:
                strengthBadge.style.color = '#e74c3c';
                strengthBadge.textContent = 'Weak';
                strengthMeter.style.backgroundColor = 'transparent';
                strengthMeter.style.width = '0';
                passwordField.style.borderColor = '#e74c3c'; // Red border color for weak password
                break;
            case 1:
                strengthBadge.style.color = '#e67e22';
                strengthBadge.textContent = 'Moderate';
                strengthMeter.style.backgroundColor = '#e67e22';
                strengthMeter.style.width = '25%';
                passwordField.style.borderColor = '#e67e22'; // Orange border color for moderate password
                break;
            case 2:
                strengthBadge.style.color = '#f1c40f';
                strengthBadge.textContent = 'Good';
                strengthMeter.style.backgroundColor = '#f1c40f';
                strengthMeter.style.width = '50%';
                passwordField.style.borderColor = '#f1c40f'; // Yellow border color for good password
                break;
            case 3:
                strengthBadge.style.color = '#2ecc71';
                strengthBadge.textContent = 'Strong';
                strengthMeter.style.backgroundColor = '#2ecc71';
                strengthMeter.style.width = '75%';
                passwordField.style.borderColor = '#2ecc71'; // Green border color for strong password
                break;
            case 4:
                strengthBadge.style.color = '#27ae60';
                strengthBadge.textContent = 'Very Strong';
                strengthMeter.style.backgroundColor = '#27ae60';
                strengthMeter.style.width = '100%';
                passwordField.style.borderColor = '#27ae60'; // Dark green border color for very strong password
                break;
        }
    });

    // Form submission
    document.getElementById('signupForm').addEventListener('submit', function (event) {
        event.preventDefault();
        var password = document.getElementById('password').value;
        var confirmPassword = document.getElementById('confirmPassword').value;

        if (password === confirmPassword) {
            var formData = new FormData(this);
            var xhr = new XMLHttpRequest();
            xhr.onreadystatechange = function () {
                if (xhr.readyState === XMLHttpRequest.DONE) {
                    if (xhr.status === 200) {
                        alert('Sign up successful!');
                        // Redirect or show success message
                    } else {
                        var errorMessage = JSON.parse(xhr.responseText).error;
                        document.getElementById('error-message').textContent = errorMessage;
                    }
                }
            };
            xhr.open('POST', 'signup.php', true);
            xhr.send(formData);
        } else {
            document.getElementById('error-message').textContent = "Oops!! Passwords do not match!";
        }
    });
</script>
